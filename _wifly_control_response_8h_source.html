39 class SimpleResponse : public WiflyResponse
40 {
41 public:
42  SimpleResponse(uint8_t cmd) : mCmd(cmd) {};
43 
44  /*
45  * Validate and convert data from response_frame
46  * @return true, if convertion was successfull, false if data was corrupted and a retry might be successfull.
47  * @throw FatalError if command code of the response doesn't match the code of the request
48  * @throw ScriptBufferFull if script buffer in PIC firmware is full and request couldn't be executed
49  */
50  bool Init(response_frame& pData, const size_t dataLength)
51  {
52  if(dataLength < 4) {
53  // response to short -> seems corrupted, allow retry
54  return false;
55  }
56 
57  if(mCmd != pData.cmd) {
58  // response doesn't match to request, allow retry
59  return false;
60  }
61 
62  switch(pData.state)
63  {
64  case OK:
65  return true;
66  case SCRIPTBUFFER_FULL:
67  throw ScriptBufferFull{};
68  case CRC_CHECK_FAILED:
69  case BAD_PACKET:
70  return false;
71  case BAD_COMMAND_CODE:
72  throw FatalError("FIRMWARE RECEIVED A BAD COMMAND CODE" + pData.cmd);
73  default:
74  throw FatalError("Unexpected response state: " + pData.state);
75  };
76  };
77 
78 private:
79  const uint8_t mCmd;
80 };
